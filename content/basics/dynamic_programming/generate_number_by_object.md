getCBS(num, n):
    b = 0
    vector<char> a
    for k = 1..(2 * n):
        if num < cnt[2 * n - k, b + 1]:
            a.push_back('('), b += 1
        else:
            num -= cnt[2 * n - k, b + 1]
            a.push_back(')'), b -= 1
    return a
